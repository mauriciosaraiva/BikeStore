public int Add(int c, int d)
{
    int sum = c + d;
    Console.WriteLine("Sum: " + sum);
    return sum;
}

public int div(int a, int b)
{
    int div = a / b;
    Console.WriteLine("Div: " + div);
    return div;
}

public int Sub(int c, int d)
{
    int sub = c - d;
    Console.WriteLine("Sub: " + sub);
    return sub;
}
